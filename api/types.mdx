This page documents all TypeScript interfaces and types exported by the React Voice Visualizer library.

## Main Types

### Controls

The main interface returned by the `useVoiceVisualizer` hook. Contains all state values and control functions.

```tsx
interface Controls {
  // Refs
  audioRef: MutableRefObject<HTMLAudioElement | null>;
  
  // Recording State
  isRecordingInProgress: boolean;
  isPausedRecording: boolean;
  recordingTime: number;
  formattedRecordingTime: string;
  isProcessingStartRecording: boolean;
  
  // Audio Data
  audioData: Uint8Array;
  recordedBlob: Blob | null;
  audioSrc: string;
  bufferFromRecordedBlob: AudioBuffer | null;
  mediaRecorder: MediaRecorder | null;
  
  // Playback State
  isPausedRecordedAudio: boolean;
  currentAudioTime: number;
  duration: number;
  formattedDuration: string;
  formattedRecordedAudioCurrentTime: string;
  
  // Availability & Processing
  isAvailableRecordedAudio: boolean;
  isProcessingRecordedAudio: boolean;
  isProcessingOnResize: boolean;
  isCleared: boolean;
  isPreloadedBlob: boolean;
  
  // Error
  error: Error | null;
  
  // Control Functions
  startRecording: () => void;
  togglePauseResume: () => void;
  startAudioPlayback: () => void;
  stopAudioPlayback: () => void;
  stopRecording: () => void;
  saveAudioFile: () => void;
  clearCanvas: () => void;
  setCurrentAudioTime: Dispatch<SetStateAction<number>>;
  setPreloadedAudioBlob: (blob: Blob) => void;
  
  // Internal Functions
  _setIsProcessingAudioOnComplete: Dispatch<SetStateAction<boolean>>;
  _setIsProcessingOnResize: Dispatch<SetStateAction<boolean>>;
}
```

### useVoiceVisualizerParams

Parameters accepted by the `useVoiceVisualizer` hook. All properties are optional.

```tsx
interface useVoiceVisualizerParams {
  onStartRecording?: () => void;
  onStopRecording?: () => void;
  onPausedRecording?: () => void;
  onResumedRecording?: () => void;
  onClearCanvas?: () => void;
  onEndAudioPlayback?: () => void;
  onStartAudioPlayback?: () => void;
  onPausedAudioPlayback?: () => void;
  onResumedAudioPlayback?: () => void;
  onErrorPlayingAudio?: (error: Error) => void;
  shouldHandleBeforeUnload?: boolean;
}
```

<ResponseField name="onStopRecording" type="() => void">

<ResponseField name="onPausedRecording" type="() => void">
  Callback invoked when audio recording is paused.

<ResponseField name="onResumedRecording" type="() => void">
  Callback invoked when audio recording is resumed.

<ResponseField name="onClearCanvas" type="() => void">

<ResponseField name="onEndAudioPlayback" type="() => void">

  Callback invoked when recorded audio playback starts.

## Internal Types

These types are used internally by the library and are exported for advanced use cases.

### BarItem

Represents a single bar in the waveform visualization.

```tsx
interface BarItem {
  startY: number;
  barHeight: number;
}
```

<ResponseField name="startY" type="number">
  The Y coordinate where the bar starts on the canvas.
</ResponseField>

<ResponseField name="barHeight" type="number">
  The height of the bar in pixels.
</ResponseField>

### BarsData

Data structure returned by the audio processing worker.

```tsx
interface BarsData {
  max: number;
}
```

<ResponseField name="max" type="number">
  The maximum amplitude value for the bar.
</ResponseField>

### GetBarsDataParams

Parameters passed to the Web Worker for processing audio buffer data.

```tsx
type GetBarsDataParams = {
  bufferData: Float32Array;
  height: number;
  width: number;
  barWidth: number;
  gap: number;
};
```

<ResponseField name="bufferData" type="Float32Array">
  The raw audio buffer data from the AudioBuffer.
</ResponseField>

<ResponseField name="height" type="number">
  The height of the canvas in pixels.
</ResponseField>

<ResponseField name="width" type="number">
  The width of the canvas in pixels.
</ResponseField>

<ResponseField name="barWidth" type="number">
  The width of each bar in pixels.
</ResponseField>

<ResponseField name="gap" type="number">
  The gap between bars.
</ResponseField>

### UseWebWorkerParams

Generic interface for the Web Worker hook.

```tsx
interface UseWebWorkerParams<T> {
  fn: AnyFunction;
  initialValue: T;
  onMessageReceived?: () => void;
}
```

<ResponseField name="fn" type="AnyFunction">
  The function to run in the Web Worker.
</ResponseField>

<ResponseField name="initialValue" type="T">
  The initial value for the result state.
</ResponseField>

<ResponseField name="onMessageReceived" type="() => void">
  Callback invoked when the worker sends a message.
</ResponseField>

### AnyFunction

Utility type for any function.

```tsx
type AnyFunction = (...args: any[]) => any;
```

## Usage Examples

### Type-Safe Hook Usage

```tsx
import { useVoiceVisualizer, useVoiceVisualizerParams } from '@gideo-llc/react-voice-visualizer';

const params: useVoiceVisualizerParams = {
  onStartRecording: () => console.log('Started'),
  onStopRecording: () => console.log('Stopped'),
  shouldHandleBeforeUnload: true,
};

const controls = useVoiceVisualizer(params);
```

### Using Controls Type

```tsx
import { Controls } from '@gideo-llc/react-voice-visualizer';
import { createContext } from 'react';

const AudioContext = createContext<Controls | null>(null);

function AudioProvider({ children }: { children: React.ReactNode }) {
  const controls = useVoiceVisualizer();
  
  return (
    <AudioContext.Provider value={controls}>
      {children}
    </AudioContext.Provider>
  );
}
```

### Custom Component with Controls

```tsx
import { Controls } from '@gideo-llc/react-voice-visualizer';

interface CustomPlayerProps {
  audioControls: Controls;
  className?: string;
}

function CustomPlayer({ audioControls, className }: CustomPlayerProps) {
  return (
    <div className={className}>
      <button onClick={audioControls.startRecording}>
        Start
      </button>
      <button onClick={audioControls.stopRecording}>
        Stop
      </button>
      {audioControls.error && (
        <p>Error: {audioControls.error.message}</p>
      )}
    </div>
  );
}
```
